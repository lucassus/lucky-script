<program> =
  | End
  | <statements> End

<statements> = <statement> { "\n" <statement> }

<statement> =
  | "\n" <statement>
  | <comment>
  | <function-declaration>
  | <expression>
  | <block>

TODO: Add <function-expression> without <identifier>
<function-declaration> = "function" <identifier> "(" ")" <block>

<block> =
  | "{" "}"
  | "{" <statements> "}"

<expression> =
  | <assigment>
  | <term> { "+" | "-" <term> }

<assigment> = <identifier> "=" <expression>

<term> = <factor> { "*" | "/" <factor> }

<factor> =
  | "+" | "-" <factor>
  | <power>

<power> = <atom> { "**" <factor> }

<atom> =
  | <number>
  | <function-call>
  | <identifier>
  | <group>

<function-call> = <identifier> "(" ")"

<group> = "(" <expression> ")"
